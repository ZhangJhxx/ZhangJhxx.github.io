6. 代码题：

   要求：

   1. m 块钱随机分配给 n 个人

   2. max: m/2 min:0.1

   ```js
   function coupon(m, n) {
     let remain = m - 0.1 * n;
     const res = new Array(n).fill(0.1);
     for (let i = 0; i < n - 1; i++) {
       let random = remain * Math.random().toFixed(1); //string
       if (random > m / 2 - 0.1) {
         random = m / 2 - 0.1;
       }
       if(random > remain){
         random = remain;
         res[i] += random;
         break;
       }
       res[i] += random;
       remain -= random;
     }
     res[n - 1] += remain;
     return res;
   }
   ```
